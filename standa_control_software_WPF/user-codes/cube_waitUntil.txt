width = 500;
length = 500;
height = 100;

kiekx = 10;
kiekz = 3;

moveA.Speed = 1000;


for( k=1; k<kiekz; k++)
{
	for( i=1; i<kiekx; i++)
	{
		direction = (-1)^i;
		
		if(direction == 1)
		{
			//moveA.Shutter(true);
			shutter("s",true);
		}
		else
		{
			//moveA.Shutter(false);
			shutter("s",false);
		}

		xas = i*width/kiekx - width/2;
		yas = direction*length/2;
		zas = height/kiekz*k;
		moveA("xyz", xas, yas, zas, length/2,  0, 0);
		
	}
}
