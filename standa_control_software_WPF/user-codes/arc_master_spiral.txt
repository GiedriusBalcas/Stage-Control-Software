

moveA.Speed = 200.01;
moveA.Line = true;

dl = 10;
radiusStart = 50;
radiusEnd = 100;

kiek2pi = 3;
maxAngle  = 2*PI*kiek2pi;
minAngle = 0;

theta = minAngle;

while(theta <= maxAngle)
{
	radius = radiusStart + (radiusEnd-radiusStart) / maxAngle * theta;
	
	dtheta = asin(dl/radius);
	
	xas = sin(theta)*radius;
	yas = cos(theta)*radius;
	moveA("xy", xas, yas);	
	
	theta += dtheta;
}
