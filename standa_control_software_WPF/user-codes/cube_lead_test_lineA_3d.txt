width = 100;
length = 100;
height = 100;

kiekx = 1000;
kiekz = 10;

set("s","DelayOn",1);
set("s","DelayOff",1);

lineA.Speed = 1000.01;
lineA.LeadIn = true;
lineA.LeadOut = true;
lineA.Shutter = true;
set("xy","DefaultSpeed",1000);

y_prev = 0;
for( k=0; k<kiekz; k++)
{
	zas = height/kiekz*k;
	jumpA("z",zas);
	for( i=0; i<kiekx; i++)
	{
		direction = (-1)^i;
		
		xas = i*width/kiekx - width/2;
		yas = direction*length/2;
		
		lineA("xy", xas, y_prev, xas, yas);
		y_prev = yas;
	}
}
