width = 25;
length = 500;
height = 100;

kiekx = 10;
kiekz = 2;

moveA.Speed = 1000.01;
moveA.Line = true;

moveA.LeadIn = false;
moveA.LeadOut = false;
moveA.Shutter = false;

for( k=0; k<kiekz; k++)
{
	
	zas = height/kiekz*k;
	xas = 0.1;
	yas = 0.1;
	moveA("xyz", xas, yas, zas);

	for( i=0; i<kiekx; i++)
	{
		direction = (-1)^i;
		
		xas = i*width/kiekx - width/2;
		yas = direction*length/2;


		moveA("x", xas);
		
		moveA.LeadIn = true;
		moveA.LeadOut = true;
		moveA.Shutter = true;

		moveA("y", yas);
		moveA.Shutter = false;
		moveA.LeadIn = false;
		moveA.LeadOut = false;
		
	}
}
