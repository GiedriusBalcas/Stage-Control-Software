radius = 500;
dl = 300;
dtheta = tan(dl/radius);

theta_max = 2*PI;

moveA.Speed = 1000;


xas_prev = sin(0) * radius;
yas_prev = cos(0) * radius;
moveA("xyz", xas_prev, yas_prev, 0);


for( theta = dtheta; theta <= theta_max; theta += dtheta)
{
	xas = sin(theta) * radius;
	yas = cos(theta) * radius;
	zas = 0;
	
	xas_prolong = xas + (xas - xas_prev) 
	yas_prolong = yas + (yas - yas_prev) 
	
	moveA("xy", xas_prolong, yas_prolong);

	xas_prev = xas;
	yas_prev = yas;
}

moveA("xy", xas, yas);
