width = 500;
length = 150;
height = 100;

kiekx = 100;
kiekz = 10;

moveA.Speed = 1000000.01;
moveA.Line = true;

for( k=0; k<kiekz; k++)
{
	for( i=0; i<kiekx; i++)
	{
		direction = (-1)^i;
		
		xas = i*width/kiekx - width/2;
		yas = direction*length/2;
		zas = height/kiekz*k;
		moveA("xyz", xas, yas, zas);
		
	}
}
