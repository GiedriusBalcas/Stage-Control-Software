width = 500;
length = 150;
height = 100;

kiekx = 10;
kiekz = 5;



for( k=1; k<kiekz; k++)
{
	for( i=1; i<kiekx; i++)
	{
		direction = (-1)^i;
		if(direction == 1)
		{
			shutter("s", 0);
		}
		else
		{
			shutter("s", 1);

		}

		xas = i*width/kiekx - width/2;
		yas = direction*length/2;
		zas = height/kiekz*k;
		moveA("xyz", xas, yas, zas);
		
	}
}
