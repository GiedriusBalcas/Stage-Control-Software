width = 500;
length = 150;
height = 100;

kiekx = 1000;
kiekz = 100;

moveA.Speed = 100000000;


for( k=1; k<kiekz; k++)
{
	for( i=1; i<kiekx; i++)
	{
		direction = (-1)^i;
		
		if(direction == 1)
		{
			moveA.Shutter(true);
		}
		else
		{
			moveA.Shutter(false);
		}

		xas = i*width/kiekx - width/2;
		yas = direction*length/2;
		zas = height/kiekz*k;
		moveA("xyz", xas, yas, zas);
		
	}
}
